JavaScript Cheat Sheet

Function Initialization

function myFunction() { /* Function body */ }

Arrow Function

const myFunction = () => { /* Function body */ };

Variable Declaration (var)

var x = 10;

Variable Declaration (let)

let x = 10;

Variable Declaration (const)

const x = 10;

Template Literals

const message = `Hello, ${name}!`;

String Concatenation

const message = 'Hello, ' + name + '!';

Object Creation

const person = { name: 'John', age: 30 };

Object Destructuring

const { name, age } = person;

Array Creation

const numbers = [1, 2, 3, 4, 5];

Array Destructuring

const [first, second] = numbers;

Array Map

const doubled = numbers.map(num => num * 2);

Array Filter

const evens = numbers.filter(num => num % 2 === 0);

Array Reduce

const sum = numbers.reduce((acc, num) => acc + num, 0);

Promise

const promise = new Promise((resolve, reject) => { /* Async code */ });

Async/Await

async function fetchData() { const data = await fetch('url'); }

SetTimeout

setTimeout(() => { console.log('Hello'); }, 1000);

SetInterval

setInterval(() => { console.log('Hello'); }, 1000);

Event Listener

element.addEventListener('click', () => { /* Handle click */ });

Event Delegation

container.addEventListener('click', (event) => { if (event.target.matches('.item')) { /* Handle item click */ } });

Local Storage

localStorage.setItem('key', 'value'); const value = localStorage.getItem('key');

Session Storage

sessionStorage.setItem('key', 'value'); const value = sessionStorage.getItem('key');

JSON Parsing

const obj = JSON.parse('{"key": "value"}');

JSON Stringify

const str = JSON.stringify({ key: 'value' });

Fetch API

fetch('url').then(response => response.json()).then(data => console.log(data));

XMLHttpRequest

const xhr = new XMLHttpRequest(); xhr.open('GET', 'url', true); xhr.onload = () => { if (xhr.status === 200) { console.log(xhr.responseText); } }; xhr.send();

Destructuring Assignment

const { a, b } = { a: 1, b: 2 };

Spread Operator

const arr = [1, 2, 3]; const newArr = [...arr, 4, 5];

Rest Parameters

function sum(...numbers) { return numbers.reduce((acc, num) => acc + num, 0); }

Class Definition

class Person { constructor(name) { this.name = name; } greet() { console.log('Hello, ' + this.name); } }

Inheritance

class Animal { constructor(name) { this.name = name; } } class Dog extends Animal { bark() { console.log('Woof!'); } }

Static Methods

class MathUtils { static add(a, b) { return a + b; } }

Symbol Creation

const sym = Symbol('description');

Generators

function* generator() { yield 1; yield 2; } const gen = generator(); console.log(gen.next().value);

WeakMap

const wm = new WeakMap(); const obj = {}; wm.set(obj, 'value');

WeakSet

const ws = new WeakSet(); const obj = {}; ws.add(obj);

Set

const set = new Set([1, 2, 2, 3]); set.add(4);

Map

const map = new Map(); map.set('key', 'value'); const value = map.get('key');

Promise.all

Promise.all([promise1, promise2]).then(results => { /* Handle results */ });

Promise.race

Promise.race([promise1, promise2]).then(result => { /* Handle result */ });
